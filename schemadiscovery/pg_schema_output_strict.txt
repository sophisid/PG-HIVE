-- NODE TYPES --
CREATE NODE TYPE PersonType : Person {firstName STRING, lastName STRING, age INTEGER};
CREATE NODE TYPE PostType : Post {content STRING, creationDate DATE};
CREATE NODE TYPE CommentType : Comment {content STRING, creationDate DATE};
CREATE NODE TYPE OrganizationType : Organization {name STRING, location STRING};

-- EDGE TYPES --
CREATE EDGE TYPE knowsType : KNOWS;
CREATE EDGE TYPE likesType : LIKES;
CREATE EDGE TYPE worksAtType : WORKS_AT;

CREATE GRAPH TYPE SmallGraphSchema STRICT {
  (PersonType),
  (PostType),
  (CommentType),
  (OrganizationType),

  (:PersonType)-[knowsType]->(:PersonType),
  (:PersonType)-[likesType]->(:PostType|CommentType),
  (:PersonType)-[worksAtType]->(:OrganizationType),

  // Constraints
  FOR (x:PersonType) SINGLETON y WITHIN (x)-[y: knowsType]->(:PersonType)
  FOR (x:PersonType) SINGLETON y WITHIN (x)-[y: likesType]->(:PostType|CommentType)
  FOR (x:PersonType) SINGLETON y WITHIN (x)-[y: worksAtType]->(:OrganizationType)
}
