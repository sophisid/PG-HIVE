CREATE GRAPH TYPE NewGraphSchema LOOSE {
  (Organisation_PlaceType: Organisation | Place {type STRING, name STRING, id DOUBLE, url STRING}),
  (PersonType: Person {browserUsed STRING, lastName STRING, firstName STRING, gender STRING, id DOUBLE, locationIP STRING, creationDate DATE, birthday DATE}),
  (EmailType: Email {address STRING}),
  (:PersonType)-[HAS_EMAILType: HAS_EMAIL]->(:EmailType),
  (:PersonType)-[IS_LOCATED_INType: IS_LOCATED_IN]->(:PlaceType),
  (:PlaceType|OrganisationType)-[IS_LOCATED_IN_IS_PART_OFType: IS_LOCATED_IN | IS_PART_OF]->(:PlaceType),
  (:PersonType)-[SPEAKSType: SPEAKS {language STRING}]->(:PersonType),
  (:PersonType)-[STUDY_ATType: STUDY_AT {classYear INTEGER}]->(:OrganisationType),
  (:PersonType)-[WORKS_ATType: WORKS_AT {workFrom INTEGER}]->(:OrganisationType),
}
