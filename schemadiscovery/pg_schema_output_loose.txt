CREATE GRAPH TYPE SmallGraphSchema LOOSE {
  (PersonType: Person {firstName STRING, lastName STRING, age INTEGER}),
  (PostType: Post {content STRING, creationDate DATE}),
  (CommentType: Comment {content STRING, creationDate DATE}),
  (OrganizationType: Organization {name STRING, location STRING}),

  (:PersonType)-[knowsType: KNOWS]->(:PersonType),
  (:PersonType)-[likesType: LIKES]->(:PostType|CommentType),
  (:PersonType)-[worksAtType: WORKS_AT]->(:OrganizationType)
}
