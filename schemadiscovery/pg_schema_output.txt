CREATE GRAPH TYPE NewGraphSchema LOOSE {
  (PersonType: Person {browserUsed STRING, lastName STRING, firstName STRING, gender STRING, id DOUBLE, locationIP STRING, creationDate DATE, birthday DATE}),
  (PostType: Post {browserUsed STRING, id DOUBLE, locationIP STRING, length INTEGER, creationDate DATE, content STRING, language STRING, imageFile STRING}),
  (TagType: Tag {name STRING, id DOUBLE, url STRING}),
  (OrganisationType: Organisation {type STRING, name STRING, id DOUBLE, url STRING}),
  (CommentType: Comment {browserUsed STRING, content STRING, id DOUBLE, locationIP STRING, length INTEGER, creationDate DATE}),
  (ForumType: Forum {id DOUBLE, title STRING, creationDate DATE}),
  (:ForumType)-[CONTAINER_OF: container_of]->(:PostType),
  (:Person_PostType)-[HAS_CREATOR: has_creator]->(:Person_CommentType),
  (:PersonType)-[HAS_INTEREST: has_interest]->(:TagType),
  (:ForumType)-[HAS_MEMBER: has_member {joinDate DATE}]->(:PersonType),
  (:ForumType)-[HAS_MODERATOR: has_moderator]->(:PersonType),
  (:Forum_Comment_PostType)-[HAS_TAG: has_tag]->(:TagType),
  (:TagType)-[HAS_TYPE: has_type]->(:TagClassType),
  (:Comment_PostType)-[IS_LOCATED_IN: is_located_in]->(:PlaceType),
  (:PersonType)-[KNOWS: knows {creationDate DATE}]->(:PersonType),
  (:PersonType)-[LIKES: likes {creationDate DATE}]->(:Comment_PostType),
  (:CommentType)-[REPLY_OF: reply_of]->(:Comment_PostType),
  (:PersonType)-[WORKS_AT: works_at {workFrom INTEGER}]->(:OrganisationType),
}
