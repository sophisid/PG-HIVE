CREATE GRAPH TYPE NewGraphSchema LOOSE{
  (PersonType: Person {BROWSERUSED STRING, LASTNAME STRING, FIRSTNAME STRING, GENDER STRING, ID DOUBLE, LOCATIONIP STRING, CREATIONDATE DATE, BIRTHDAY DATE}),
  (PostType: Post {BROWSERUSED STRING, CONTENT STRING, ID DOUBLE, LOCATIONIP STRING, LENGTH INTEGER, LANGUAGE STRING, CREATIONDATE DATE}),
  (TagType: Tag {NAME STRING, ID DOUBLE, URL STRING}),
  (OrganisationType: Organisation {TYPE STRING, NAME STRING, ID DOUBLE, URL STRING}),
  (CommentType: Comment {BROWSERUSED STRING, CONTENT STRING, ID DOUBLE, LOCATIONIP STRING, LENGTH INTEGER, CREATIONDATE DATE}),
  (Forum_PostType: Post {ID DOUBLE, TITLE STRING, CREATIONDATE DATE, BROWSERUSED STRING, LOCATIONIP STRING, IMAGEFILE STRING, LENGTH INTEGER}),
  (:ForumType)-[CONTAINER_OF: container_of]->(:PostType),
  (:PersonType)-[HAS_CREATOR: has_creator]->(:CommentType),
  (:Post_ForumType)-[HAS_CREATOR_HAS_MODERATOR_HAS_TAG: has_creator_has_moderator_has_tag]->(:Tag_PersonType),
  (:PersonType)-[HAS_INTEREST: has_interest]->(:TagType),
  (:ForumType)-[HAS_MEMBER: has_member {JOINDATE DATE}]->(:PersonType),
  (:PostType)-[HAS_TAG_IS_LOCATED_IN: has_tag_is_located_in]->(:Tag_PlaceType),
  (:Place_Comment_PersonType)-[HAS_TAG_IS_PART_OF_REPLY_OF_STUDIES_AT: has_tag_is_part_of_reply_of_studies_at {CLASSYEAR INTEGER}]->(:Tag_Place_Comment_OrganisationType),
  (:Tag_CommentType)-[HAS_TYPE_IS_LOCATED_IN: has_type_is_located_in]->(:TagClass_PlaceType),
  (:PersonType)-[KNOWS: knows {CREATIONDATE DATE}]->(:PersonType),
  (:PersonType)-[LIKES: likes {CREATIONDATE DATE}]->(:Comment_PostType),
  (:CommentType)-[REPLY_OF: reply_of]->(:PostType),
  (:PersonType)-[WORKS_AT: works_at {WORKFROM INTEGER}]->(:OrganisationType),
}
